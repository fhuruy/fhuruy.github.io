# remove the outliers

def remove_outliers(feature,data):
    first_quartile = data[feature].describe()['25%']
    third_quartile = data[feature].describe()['75%']
    iqr = third_quartile - first_quartile
    
    index = data[(data[feature] < (first_quartile - 3*iqr)) | (data[feature] > (first_quartile + 3*iqr))].index
    data = data.drop(index)
    return data
outlier_features = ['Insulin', 'Glucose', 'BloodPressure', 'SkinThickness', 'BMI']
for feat in outlier_features:
    data = remove_outliers(feat,data)
    
def knn_method():
    # replace missing data with midean
    values = {'Glucose': data['Glucose'].median(),'BloodPressure':data['BloodPressure'].median(),'BMI':data['BMI'].median()}
    data.fillna(value=values,inplace=True)
 
    # using KNNImputer fill Insulin SkinThickness
    corr_SkinThickness = ['BMI', 'Glucose','BloodPressure', 'SkinThickness']
   
    SkinThickness_imputer = KNNImputer(weights = 'distance')
    data[corr_SkinThickness] = SkinThickness_imputer.fit_transform(data[corr_SkinThickness])
 
    corr_Insulin = ['Glucose', 'BMI','BloodPressure', 'Insulin']
    Insulin_imputer = KNNImputer(weights = 'distance')
    data[corr_Insulin] = Insulin_imputer.fit_transform(data[corr_Insulin])
    
    
knn_method()
